## Overview

The `ChatInputArea` component provides a rich input interface for users to interact with the chat system. It includes text input, file attachments (images and PDFs), voice input, and quick action tools.

## Features

- **Textarea Input**: Auto-resizing textarea for text messages
- **File Attachments**: Drag-and-drop support for images and PDFs
- **Voice Input**: Integrated voice button with dual-mode support (Gemini AI / native)
- **Quick Actions**: Dropdown menu with AI-powered shortcuts
- **Global Drop Zone**: Drop files anywhere on the page
- **Keyboard Shortcuts**: Submit with Cmd/Ctrl+Enter
- **Status-Aware**: Disabled during processing states

## Props

```typescript
interface ChatInputAreaProps {
  /** Reference to the textarea element for programmatic control */
  textareaRef: React.RefObject<HTMLTextAreaElement | null>;
  
  /** Current input value */
  input: string;
  
  /** Callback when input text changes */
  onInputChange: (e: React.ChangeEvent<HTMLTextAreaElement>) => void;
  
  /** Callback when message is submitted */
  onSubmit: (message: PromptInputMessage) => void;
  
  /** Whether sending is allowed (based on chat status) */
  canSend: boolean;
  
  /** Current chat status */
  status: 'ready' | 'streaming' | 'submitted' | 'error';
  
  /** Whether a file is being analyzed */
  isAnalyzingFile: boolean;
  
  /** Props to pass to the VoiceButton component */
  voiceProps: VoiceButtonProps;
  
  /** Optional callback for quick action selection */
  onQuickAction?: (prompt: string) => void;
}
```

### VoiceButtonProps

```typescript
interface VoiceButtonProps {
  /** Whether voice recording is currently active */
  isListening: boolean;
  
  /** Whether voice is being processed by AI */
  isProcessing: boolean;
  
  /** Whether voice input is supported in the browser */
  isSupported: boolean;
  
  /** Current voice recognition mode */
  mode: 'gemini' | 'native';
  
  /** Callback to toggle voice recording */
  onClick: () => void;
  
  /** Whether the button should be disabled */
  disabled?: boolean;
}
```

### PromptInputMessage

```typescript
interface PromptInputMessage {
  /** Message text content */
  text: string;
  
  /** Attached files (images, PDFs) */
  files: File[];
}
```

## Usage

### Basic Example

```tsx
import { ChatInputArea } from '@/app/components/features/chat/chat-input-area';
import { useRef, useState } from 'react';

function ChatInterface() {
  const textareaRef = useRef<HTMLTextAreaElement>(null);
  const [input, setInput] = useState('');
  const [status, setStatus] = useState<'ready'>('ready');

  const handleSubmit = (message: PromptInputMessage) => {
    console.log('Submitting:', message.text, message.files);
    // Send message to API
  };

  return (
    <ChatInputArea
      textareaRef={textareaRef}
      input={input}
      onInputChange={(e) => setInput(e.target.value)}
      onSubmit={handleSubmit}
      canSend={status === 'ready' && input.trim().length > 0}
      status={status}
      isAnalyzingFile={false}
      voiceProps={{
        isListening: false,
        isProcessing: false,
        isSupported: true,
        mode: 'gemini',
        onClick: () => console.log('Toggle voice'),
        disabled: false,
      }}
    />
  );
}
```

### With Voice Input

```tsx
import { useVoiceInput } from '@/app/hooks/use-voice-input';

function ChatWithVoice() {
  const textareaRef = useRef<HTMLTextAreaElement>(null);
  const [input, setInput] = useState('');

  const updateTextareaValue = useCallback((value: string) => {
    const textarea = textareaRef.current;
    if (textarea) {
      const nativeInputValueSetter = Object.getOwnPropertyDescriptor(
        window.HTMLTextAreaElement.prototype,
        'value'
      )?.set;
      nativeInputValueSetter?.call(textarea, value);
      const event = new Event('input', { bubbles: true });
      textarea.dispatchEvent(event);
    }
    setInput(value);
  }, []);

  const {
    isListening,
    isProcessing,
    isSupported,
    mode,
    toggleListening,
  } = useVoiceInput({ onTranscript: updateTextareaValue });

  return (
    <ChatInputArea
      textareaRef={textareaRef}
      input={input}
      onInputChange={(e) => setInput(e.target.value)}
      onSubmit={handleSubmit}
      canSend={true}
      status="ready"
      isAnalyzingFile={false}
      voiceProps={{
        isListening,
        isProcessing,
        isSupported,
        mode,
        onClick: toggleListening,
        disabled: false,
      }}
    />
  );
}
```

### With Quick Actions

```tsx
import { QUICK_ACTIONS } from '@/app/components/features/chat/chat-quick-actions';

function ChatWithQuickActions() {
  const handleQuickAction = useCallback((prompt: string) => {
    const action = QUICK_ACTIONS.find((a) => a.prompt === prompt);

    if (action?.formFields?.length > 0) {
      // Show inline form
      setActiveQuickAction(action);
    } else if (!prompt.endsWith(' ')) {
      // Complete prompt - send directly
      handleSubmit({ text: prompt, files: [] });
    } else {
      // Incomplete prompt - populate input
      setInput(prompt);
      textareaRef.current?.focus();
    }
  }, []);

  return (
    <ChatInputArea
      // ... other props
      onQuickAction={handleQuickAction}
    />
  );
}
```

## Component Structure

The ChatInputArea uses the `PromptInput` primitive components:

```tsx

  globalDrop
  multiple={false}
  maxFiles={1}
  accept="image/*,application/pdf"

      {(attachment) => <PromptInputAttachment data={attachment} />}

      ref={textareaRef}
      onChange={onInputChange}
      placeholder="Escribe tu pregunta o sube imágenes/PDFs..."
    />
  </PromptInputBody>

  <PromptInputFooter>
    <PromptInputTools>
      {/* Quick Actions Menu */}
      <PromptInputActionMenu>
        <PromptInputActionMenuTrigger />
        <PromptInputActionMenuContent>
          {/* Quick action items */}
          <PromptInputActionAddAttachments />
        </PromptInputActionMenuContent>
      </PromptInputActionMenu>

      {/* Voice Button */}

    {/* Submit Button */}

  </PromptInputFooter>
```

## Integration Example

Complete integration with all features:

```tsx
function CompleteChat() {
  const textareaRef = useRef<HTMLTextAreaElement>(null);
  const [input, setInput] = useState('');
  const [activeQuickAction, setActiveQuickAction] = useState<QuickAction | null>(null);

  const { messages, sendMessage, status } = usePersistentChat();
  const { handleSubmit, isAnalyzing } = useFileSubmission({
    setMessages,
    sendMessage,
  });

  const updateTextareaValue = useCallback((value: string) => {
    const textarea = textareaRef.current;
    if (textarea) {
      const nativeInputValueSetter = Object.getOwnPropertyDescriptor(
        window.HTMLTextAreaElement.prototype,
        'value'
      )?.set;
      nativeInputValueSetter?.call(textarea, value);
      const event = new Event('input', { bubbles: true });
      textarea.dispatchEvent(event);
    }
    setInput(value);
  }, []);

  const voiceInput = useVoiceInput({ onTranscript: updateTextareaValue });

  const canSend = (status === 'ready' || status === 'error') && !isAnalyzing;

  const handleQuickAction = useCallback((prompt: string) => {
    const action = QUICK_ACTIONS.find((a) => a.prompt === prompt);
    if (action?.formFields?.length > 0) {
      setActiveQuickAction(action);
    } else if (!prompt.endsWith(' ')) {
      handleSubmit({ text: prompt, files: [] });
    } else {
      updateTextareaValue(prompt);
      textareaRef.current?.focus();
    }
  }, [handleSubmit, updateTextareaValue]);

  return (
    <div>
      {/* Messages */}
      <ChatConversation messages={messages} {...props} />

      {/* Quick Action Form (if active) */}
      {activeQuickAction && (
        <QuickActionDataForm
          action={activeQuickAction}
          onSubmit={(composedPrompt) => {
            setActiveQuickAction(null);
            handleSubmit({ text: composedPrompt, files: [] });
          }}
          onCancel={() => setActiveQuickAction(null)}
        />
      )}

      {/* Input Area */}
      <ChatInputArea
        textareaRef={textareaRef}
        input={input}
        onInputChange={(e) => setInput(e.target.value)}
        onSubmit={handleSubmit}
        canSend={canSend}
        status={status}
        isAnalyzingFile={isAnalyzing}
        voiceProps={{
          isListening: voiceInput.isListening,
          isProcessing: voiceInput.isProcessing,
          isSupported: voiceInput.isSupported,
          mode: voiceInput.mode,
          onClick: voiceInput.toggleListening,
          disabled: !canSend,
        }}
        onQuickAction={handleQuickAction}
      />
    </div>
  );
}
```
